00155   }

00311     double orig = header_.domainorig(i);
00312     double scal = header_.domainscal(i);
00313     if(region[i] > orig+(1./scal)) {
00314       return 0;
00315     }
00316     if(region[i+dimp] < orig) {
00317       return 0;
00318     }

00330    * To traverse a non-empty binary tree in preorder,
00331    * perform the following operations recursively at each node, starting with the root node:
00332    * > visit the root;
00333    * > traverse the left subtree;

00346      * This function works when we have either points or boxes.

00364      */
